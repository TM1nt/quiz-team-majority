    <div id="teams-container"></div>

    <button onclick="calculateMajority()">多数決を実行</button>

    <div id="result" class="result"></div>


    <script>

        const teamsCount = 8; // チーム数

        const membersPerTeam = 6; // 1チームの人数

        const choices = [1, 2, 3, 4]; // 選択肢


        // チームとメンバーのUIを生成する

        function createTeamUI() {

            const container = document.getElementById("teams-container");

            for (let team = 1; team <= teamsCount; team++) {

                const teamDiv = document.createElement("div");

                teamDiv.className = "team-container";

                teamDiv.id = `team-${team}`;


                teamDiv.innerHTML = `<h2>チーム ${team}</h2>`;

                for (let member = 1; member <= membersPerTeam; member++) {

                    const memberDiv = document.createElement("div");

                    memberDiv.className = "person";

                    memberDiv.innerHTML = `

                        メンバー${member}:

                        <select id="team${team}-member${member}">

                            ${choices.map(choice => `<option value="${choice}">${choice}</option>`).join("")}

                        </select>

                    `;

                    teamDiv.appendChild(memberDiv);

                }

                container.appendChild(teamDiv);

            }

        }


        // 多数決で選択肢を決定する関数

        function calculateMajority() {

            let resultText = "";


            for (let team = 1; team <= teamsCount; team++) {

                const votes = {}; // 投票カウント

                choices.forEach(choice => votes[choice] = 0);


                for (let member = 1; member <= membersPerTeam; member++) {

                    const selectElement = document.getElementById(`team${team}-member${member}`);

                    const vote = selectElement.value;

                    votes[vote]++;

                }


                // 多数決で最も票が多い選択肢を決定

                let majorityChoice = choices[0];

                let maxVotes = 0;


                for (const choice in votes) {

                    if (votes[choice] > maxVotes) {

                        maxVotes = votes[choice];

                        majorityChoice = choice;

                    }

                }


                resultText += `チーム ${team} の多数決の結果: 選択肢 ${majorityChoice}（${maxVotes}票）<br>`;

            }


            document.getElementById("result").innerHTML = resultText;

        }


        // ページ読み込み時にUIを生成

        window.onload = createTeamUI;

    </script>
